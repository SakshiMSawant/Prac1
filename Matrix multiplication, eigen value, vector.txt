Code - eigen values and eigen vectors

importtensorflow astf 

# Eigen decomposition

A = tf.constant([[4, 2], [1, 3]], dtype=tf.float64)

eigenvalues,eigenvectors=tf.linalg.eig(A)

print(eigenvalues.numpy())

print(eigenvectors.numpy())
